CREATE TABLE FUNCIONARIO (
  CODFUNC NUMBER(5)constraint PK_FUNC primary KEY,
  NOMEFUNC VARCHAR2(40) NOT NULL,
  SALARIO NUMBER(7,2)
  );

 CREATE OR REPLACE FUNCTION F_FUNCIONARIO
(V_SALARIO IN FUNCIONARIO.SALARIO%TYPE)
RETURN NUMBER
IS
V_TOTAL NUMBER;
BEGIN
SELECT COUNT(*)INTO V_TOTAL FROM FUNCIONARIO
WHERE SALARIO=V_SALARIO;
RETURN V_TOTAL;
END F_FUNCIONARIO; 

DECLARE
  CURSOR C_FUNC IS
  SELECT CODFUNC,NOMEFUNC,SALARIO FROM FUNCIONARIO;
  V_FUNC C_FUNC%rowtype;
  begin
     open C_FUNC;
     loop 
     fetch C_FUNC into V_FUNC;
     exit when C_FUNC%notfound;
     dbms_output.put_line('FUNCIONARIO'||V_FUNC.NOMEFUNC);
     dbms_output.put_line('FUNCIONARIO'||V_FUNC.SALARIO);
     END LOOP;
     CLOSE C_FUNC;
     END;
     
     
  CREATE OR REPLACE PROCEDURE INSERIR(P_CODFUNC IN NUMBER,P_NOMEFUNC IN VARCHAR2,
  P_SALARIO IN NUMBER
 )
  IS
  BEGIN
  INSERT INTO FUNCIONARIO
    (CODFUNC,NOMEFUNC,SALARIO)
    values(P_CODFUNC,P_NOMEFUNC,P_SALARIO);
    COMMIT;
  END INSERIR;
